#include <iostream>
using namespace std;

int main()
{
    int side;

    cout << "Enter a number: ";
    cin >> side;
    int count = 0;
    while (count < side)
    {
        for (int j = count; j >= 0; j--)
        {
            cout << "#";
        }
        cout << "\n";
        count ++;
    }
}
